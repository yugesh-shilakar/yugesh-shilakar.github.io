Policy Register dApp

First, we need a smart contract that stores the policies. Here’s a simple Solidity smart contract for this purpose:

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract PolicyRegistry {
    struct Policy {
        uint256 id;
        string data;
        address owner;
    }

    Policy[] public policies;

    function registerPolicy(string memory data) public {
        uint256 id = policies.length;
        policies.push(Policy(id, data, msg.sender));
    }

    function getPolicy(uint256 id) public view returns (Policy memory) {
        require(id < policies.length, "Policy does not exist");
        return policies[id];
    }

    function getPoliciesCount() public view returns (uint256) {
        return policies.length;
    }
}

Create ASP.NET MVC project and create a model named Block.cs. I am using .NET version 8.0

using System;
using System.Security.Cryptography;
using System.Text;

namespace BlockchainApp.Models
{
    public class Block
    {
        public int Index { get; set; }
        public string PolicyData { get; set; }
        public DateTime Timestamp { get; set; }
        public string PreviousHash { get; set; }
        public string Hash { get; set; }

        public Block(int index, string policyData, string previousHash)
        {
            Index = index;
            PolicyData = policyData;
            Timestamp = DateTime.UtcNow;
            PreviousHash = previousHash;
            Hash = CalculateHash();
        }

        public string CalculateHash()
        {
            using (SHA256 sha256 = SHA256.Create())
            {
                var rawData = $"{Index}{PolicyData}{Timestamp}{PreviousHash}";
                byte[] bytes = sha256.ComputeHash(Encoding.UTF8.GetBytes(rawData));
                return BitConverter.ToString(bytes).Replace("-", "").ToLower();
            }
        }
    }
}

Then create a class for the Blockchain i.e. Blockchain.cs

using System.Collections.Generic;

namespace BlockchainApp.Models
{
    public class Blockchain
    {
        public List<Block> Chain { get; private set; }

        public Blockchain()
        {
            Chain = new List<Block>();
            // Create the genesis block
            AddBlock("Genesis Block");
        }

        public void AddBlock(string policyData)
        {
            int index = Chain.Count;
            string previousHash = Chain.Count > 0 ? Chain[index - 1].Hash : "0";
            Block newBlock = new Block(index, policyData, previousHash);
            Chain.Add(newBlock);
        }
    }
}

Now we can create controller by:

Right-click on the Controllers folder, select Add > Controller.

Choose MVC Controller - Empty and name it PolicyController

using System.Web.Mvc;
using BlockchainApp.Models;

namespace BlockchainApp.Controllers
{
    public class PolicyController : Controller
    {
        private static Blockchain blockchain = new Blockchain();

        // GET: Policy
        public ActionResult Index()
        {
            return View(blockchain.Chain);
        }

        [HttpPost]
        public ActionResult Register(string policyData)
        {
            if (!string.IsNullOrEmpty(policyData))
            {
                blockchain.AddBlock(policyData);
            }
            return RedirectToAction("Index");
        }
    }
}

Here are the key annotations for the blockchain policy registry interface shown in the image:

  • The top section shows a text input field labelled "Enter Policy Data" with a "Register" button - this is where new policies can be entered into the blockchain
  • The main table displays the blockchain data with 5 columns:
    • Index: Sequential number of each block
    • Policy Data: The actual policy information stored
    • Timestamp: When the block was created
    • Previous Hash: Hash of the previous block (0 for genesis block)
    • Hash: Unique hash for the current block
  • The table shows 4 blocks (0-3), starting with the Genesis Block and including 3 additional policy entries
  • Each block's hash is calculated using the block's data and links to the previous block's hash, forming the blockchain structure
